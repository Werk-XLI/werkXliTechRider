\newcommand{\fohInputChannels}{%
    % EINGANGSKANÄLE - Signale die von der Soundquelle auf der Bühne in das Band Stage-Rack gehen
    % Das sind z.B. Mikrofone, DI-Boxen, die auf der Bühne sind oder Talkback Mikrofone von FOH
    % Format: \rowcolor{categorycolor} Channel & Source & Microphone/Details \\
    % \hline
    %
    % Available colors: drumscolor, guitarcolor, basscolor, keyscolor, 
    %                  vocalscolor, trackscolor, monitoringcolor, systemcolor
    %
    % Example:
    % \rowcolor{drumscolor} 1 & Kick Drum & Beta 52A (Location-Kick) \\
    % \hline
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%                              IEM OUTPUT CHANNELS                             %
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\iemOutputChannels}{%
    % AUSGANGSKANÄLE - Signale die aus dem Band Stage-Rack an FOH gehen
    % Beispiele: Die Signale der Mikrofone, DI-Boxen, etc. von der Bühne, die nach dem 
    % Passiv-Splitter im Rack an FOH gehen. Oder Signale wie Backing oder Intro-Tracks, 
    % die direkt aus dem Rackmixer an FOH gehen.
    % Format: \rowcolor{monitoringcolor} Channel & Purpose & Description \\
    % \hline
    %
    % Example:
    % \rowcolor{monitoringcolor} Aux 1 & IEM Mix 1 & Schlagzeuger \\
    % \hline
}
